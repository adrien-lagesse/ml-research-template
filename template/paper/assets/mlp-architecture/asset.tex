% Architecture asset: a small MLP drawn by hand in TikZ (no data). Layers are
% input (3), two hidden (4, 4), and output (2), fully connected.
\begin{tikzpicture}[
  >=stealth, x=1.7cm, y=1.1cm,
  neuron/.style={circle, draw, thick, minimum size=7mm, inner sep=0pt},
  input/.style={neuron, fill=black!5},
  hidden/.style={neuron, fill=oiBlue!15},
  output/.style={neuron, fill=oiOrange!25},
  conn/.style={->, draw=black!30, shorten >=1pt},
]
  \foreach \i in {1,...,3} \node[input]  (i\i) at (0,{2 - \i}) {};
  \foreach \i in {1,...,4} \node[hidden] (h\i) at (1,{2.5 - \i}) {};
  \foreach \i in {1,...,4} \node[hidden] (g\i) at (2,{2.5 - \i}) {};
  \foreach \i in {1,...,2} \node[output] (o\i) at (3,{1.5 - \i}) {};

  \foreach \a in {1,...,3} \foreach \b in {1,...,4} \draw[conn] (i\a) -- (h\b);
  \foreach \a in {1,...,4} \foreach \b in {1,...,4} \draw[conn] (h\a) -- (g\b);
  \foreach \a in {1,...,4} \foreach \b in {1,...,2} \draw[conn] (g\a) -- (o\b);

  \node at (0,-2.4) {Input};
  \node at (1,-2.4) {Hidden 1};
  \node at (2,-2.4) {Hidden 2};
  \node at (3,-2.4) {Output};
\end{tikzpicture}
