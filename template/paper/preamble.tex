% Package stack for the paper. Grouped by purpose; load order matters where
% noted. Notation and \newcommand definitions live in macros.tex, not here.

\usepackage[T1]{fontenc}
\usepackage[utf8]{inputenc}

% Math. mathtools loads amsmath and fixes it; keep amssymb after.
\usepackage{mathtools}
\usepackage{amssymb}
\usepackage{bm}
\usepackage{amsthm}
\usepackage{mathrsfs}

% Numbers and units: one formatter for every number in the paper.
% separate-uncertainty prints "1.23 +- 0.04" as a mean $\pm$ std cell.
\usepackage{siunitx}
\sisetup{separate-uncertainty=true}

% Micro-typography (protrusion, and expansion where the engine allows).
\usepackage{microtype}

% Tables.
\usepackage{booktabs}
\usepackage{multirow}
\usepackage{makecell}

% Figures, sub-figures, captions. assets/ is on the graphics path so a cached
% asset PDF resolves as \includegraphics{<name>/asset}.
\usepackage{graphicx}
\graphicspath{{assets/}}
\usepackage{caption}
\usepackage{subcaption}

% General quality-of-life.
\usepackage{xcolor}
\usepackage{enumitem}
\usepackage{csquotes}

% Pseudocode: algorithmicx layout via the maintained algpseudocodex, wrapped in
% the algorithm float.
\usepackage{algorithm}
\usepackage{algpseudocodex}

% Data-driven figures and tables: one CSV feeds pgfplots (figure) and
% pgfplotstable (table). csvsimple handles bespoke, hand-shaped tables.
\usepackage{tikz}
\usepackage{pgfplots}
\pgfplotsset{compat=1.18}
\usepgfplotslibrary{fillbetween}  % shaded mean +- std bands
\usepackage{pgfplotstable}
\usepackage{csvsimple}

% Colorblind-safe categorical palette (Okabe-Ito), validated for CVD adjacent-pair
% separation. Ordered so the highest-contrast hues come first. See paper/README.md.
\definecolor{oiBlue}{HTML}{0072B2}
\definecolor{oiVermillion}{HTML}{D55E00}
\definecolor{oiGreen}{HTML}{009E73}
\definecolor{oiPurple}{HTML}{CC79A7}
\definecolor{oiOrange}{HTML}{E69F00}
\definecolor{oiSky}{HTML}{56B4E9}
\definecolor{oiYellow}{HTML}{F0E442}

% Default plot cycle: each series gets a distinct COLOR *and* line style *and*
% marker, so a figure stays readable in grayscale and for total color blindness.
\pgfplotscreateplotcyclelist{paper}{%
  {oiBlue, solid, mark=*, mark options={fill=oiBlue}},%
  {oiVermillion, densely dashed, mark=square*, mark options={fill=oiVermillion}},%
  {oiGreen, densely dashdotted, mark=triangle*, mark options={fill=oiGreen}},%
  {oiPurple, dashdotted, mark=diamond*, mark options={fill=oiPurple}},%
  {oiOrange, densely dotted, mark=pentagon*, mark options={fill=oiOrange}},%
  {oiSky, loosely dashed, mark=x},%
}
\pgfplotsset{
  cycle list name=paper,
  every axis plot/.append style={thick},
  colormap name=viridis,  % sequential magnitude: perceptually uniform, CVD-safe
}

% Citations: natbib in author-year mode gives \citet (textual) and \citep
% (parenthetical). Loaded before hyperref. Bibliography style is plainnat.
\usepackage[authoryear, round]{natbib}

% Cross-references: hyperref, then cleveref LAST so it can hook everything.
\usepackage[hidelinks]{hyperref}
\usepackage{cleveref}

% Theorem environments (declared after cleveref so it names them correctly).
\theoremstyle{plain}
\newtheorem{theorem}{Theorem}
\newtheorem{lemma}{Lemma}
\newtheorem{proposition}{Proposition}
\newtheorem{corollary}{Corollary}
\theoremstyle{definition}
\newtheorem{definition}{Definition}
